  frontend:
    build: ./frontend
    restart: unless-stopped

      - backend

  pg_data:

Backend

Tasks Table — columns ทั้งหมด:

idSERIAL PRIMARY KEY — auto-increment integer
titleVARCHAR(200) NOT NULL — ชื่องาน บังคับมี
statusVARCHAR(20) DEFAULT 'todo' — ค่าถูกต้อง: todo / in-progress / done
priorityVARCHAR(10) DEFAULT 'medium' — ค่าถูกต้อง: low / medium / high
assigneeVARCHAR(100) — ชื่อผู้รับผิดชอบ optional
due_dateDATE — วันกำหนดส่ง NULL ได้
descriptionTEXT — รายละเอียดงาน
created_at / updated_atTIMESTAMPTZ DEFAULT NOW() — กำหนดอัตโนมัติ

require.main === module guard — สำคัญสำหรับ Jest:

backend/index.js — ท้ายไฟล์
// รัน server เฉพาะเมื่อ execute โดยตรง (ไม่ใช่ require จาก test)
if (require.main === module) {
  initDb()
    .then(() => {
      app.listen(PORT, () => {
        console.log(`TaskFlow API running at http://localhost:${PORT}`);
      });
    })
    .catch(err => {
      console.error('DB init failed:', err.message);
      process.exit(1);
    });
}

module.exports = app;  // Jest import ได้โดยไม่ start server
require.main === moduleเป็น true เฉพาะตอนรัน node index.js — false ตอน Jest require('./index')
module.exports = appexport express app ให้ Jest + supertest import ได้โดยไม่ผูก port จริง

sonar-project.properties — ไฟล์สำคัญของ Week 6
SonarQube

สร้างไฟล์นี้ใน backend/ — นี่คือ ไฟล์ใหม่ที่สำคัญที่สุดใน Part 1 ซึ่งบอก SonarQube ว่าต้องวิเคราะห์อะไรและหา coverage report ที่ไหน

สร้างไฟล์ใหม่: backend/sonar-project.properties
sonar-project.properties — เนื้อหาทั้งหมด
# SonarQube Project Configuration
sonar.projectKey=taskflow
sonar.projectName=TaskFlow
sonar.projectVersion=1.0.0

sonar.sources=.
sonar.tests=tests
sonar.exclusions=node_modules/**,coverage/**,dist/**

sonar.javascript.lcov.reportPaths=coverage/lcov.info

sonar.language=js
sonar.sourceEncoding=UTF-8

อธิบายทุก property ทีละบรรทัด:

sonar.projectKey=taskflowunique ID ของโปรเจกต์ใน SonarQube dashboard — ต้องตรงกับ key ที่สร้างตอน generate SONAR_TOKEN
sonar.projectName=TaskFlowชื่อที่แสดงใน SonarQube dashboard (human-readable)
sonar.projectVersion=1.0.0version ที่แสดงใน dashboard history — ช่วย track ว่าแต่ละ version มี quality เป็นยังไง
sonar.sources=.path ของ source code ที่จะ analyze — . หมายถึง backend/ directory ทั้งหมด (เพราะ sonarqube-scan-action ใช้ projectBaseDir: backend)
sonar.tests=testsบอก SonarQube ว่า tests/ เป็น test files — metrics "production code" จะไม่นับไฟล์ใน tests/
sonar.exclusions=...ข้ามไฟล์ใน node_modules/, coverage/, dist/ — ป้องกัน dependency code ปน analyze แล้วทำให้ metrics เบี้ยว
sonar.javascript.lcov.reportPathsที่อยู่ของ coverage/lcov.info ที่ Jest สร้างด้วย --coverage — SonarQube อ่านไฟล์นี้เพื่อแสดง coverage % ใน dashboard
sonar.language=jsบอก SonarQube ว่าวิเคราะห์ในฐานะ JavaScript (ไม่ใช่ TypeScript หรือ Python)
sonar.sourceEncoding=UTF-8encoding สำหรับอ่านไฟล์ source — จำเป็นสำหรับ comment ภาษาไทยใน index.js
🔬
lcov.info คืออะไร? เมื่อรัน jest --coverage Jest จะสร้างไฟล์ coverage/lcov.info ซึ่งเป็น format มาตรฐาน Linux Coverage ที่บอกว่าแต่ละบรรทัดใน source code ถูกรันกี่ครั้งระหว่าง test

SonarQube อ่านไฟล์นี้แล้วคำนวณ coverage % แสดงใน dashboard พร้อม highlight บรรทัดที่ยังไม่ถูก test
⚠️
sonar.exclusions สำคัญมาก ถ้าไม่ exclude node_modules/** → SonarQube จะพยายาม analyze ไฟล์ใน node_modules ทั้งหมด (อาจมีหลาย 1,000 ไฟล์) → scan ช้ามาก และ metrics ผิดเพี้ยน
ถ้าไม่ exclude coverage/** → SonarQube จะ analyze ไฟล์ HTML report ที่ Jest generate ซึ่งไม่ใช่ source code จริง
📊
SonarQube วิเคราะห์อะไรบ้างจาก backend/index.js?
• Bugs — code ที่อาจทำงานผิดพลาด เช่น potential null reference • Vulnerabilities — ช่องโหว่ด้านความปลอดภัย เช่น hardcoded credentials • Code Smells — code ที่ยากอ่าน/บำรุงรักษา เช่น function ยาวเกิน, duplicate code • Coverage — % ของ statements/branches/functions ที่ถูก test ครอบคลุม • Duplications — code ที่ copy-paste ซ้ำกัน
