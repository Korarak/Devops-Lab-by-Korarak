ส่วนที่ 1 — State:

useTasks.js — State ทั้งหมด
import { ref, computed, watch } from 'vue'

export function useTasks() {

  // ── State ────────────────────────────────────────────────────────
  const tasks          = ref([])         // รายการงานจาก API
  const loading        = ref(true)       // กำลังโหลดอยู่หรือไม่
  const search         = ref('')         // ข้อความค้นหา
  const statusFilter   = ref('')         // status ที่เลือก
  const priorityFilter = ref('')         // priority ที่เลือก
  const showModal      = ref(false)      // แสดง/ซ่อน modal เพิ่ม/แก้ไข
  const editingId      = ref(null)       // null = Add, มีค่า = Edit
  const confirmDelete  = ref(null)       // task ที่กำลังจะลบ

  const emptyForm = () => ({
    title:       '',
    status:      'todo',
    priority:    'medium',
    assignee:    '',
    due_date:    '',
    description: '',
  })

  const form = ref(emptyForm())          // ข้อมูลที่ส่งให้ TaskModal
statusFilter + priorityFilter2 filter state แทนที่จะเป็น catFilter เดียว — เพิ่ม complexity ของ composable
emptyForm()function สร้าง form ว่าง — มี due_date เป็น '' (string ว่าง) แทน null เพื่อ input[type=date]

ส่วนที่ 2 — isOverdue Helper:

  // ── Helper ────────────────────────────────────────────────────────

t.due_date &&ตรวจว่ามี due_date ก่อน — ถ้าไม่มีวันกำหนดส่งไม่นับว่า overdue

ส่วนที่ 3 — Computed (3 conditions ใน filtered):

useTasks.js — Computed
  // ── Computed ─────────────────────────────────────────────────────
  const filtered = computed(() => {
    const s = search.value.toLowerCase()
    return tasks.value.filter(t => {
      const matchS        = !s || t.title.toLowerCase().includes(s) ||
                            (t.description || '').toLowerCase().includes(s)
      const matchStatus   = !statusFilter.value   || t.status   === statusFilter.value
      const matchPriority = !priorityFilter.value || t.priority === priorityFilter.value
      return matchS && matchStatus && matchPriority   // AND ทั้ง 3 conditions
    })
  })

  const stats = computed(() => ({
    total:       tasks.value.length,
    todo:        tasks.value.filter(t => t.status === 'todo').length,
    in_progress: tasks.value.filter(t => t.status === 'in-progress').length,
    done:        tasks.value.filter(t => t.status === 'done').length,
    overdue:     tasks.value.filter(t => isOverdue(t)).length,
  }))
matchS && matchStatus && matchPriorityAND ทั้ง 3 — task ต้องผ่านทุก filter จึงจะแสดง (ต่างจาก StockPro ที่มีแค่ 2)
!statusFilter.valueถ้าไม่เลือก filter → condition เป็น true เสมอ → แสดงทุก task
overdue: tasks.value.filter(...)นับงานที่ overdue จาก isOverdue helper — ใช้ใน StatsGrid แสดง warning

ส่วนที่ 4 — Methods (fetchTasks, openAdd, openEdit, saveTask, deleteTask):

useTasks.js — API Methods
  async function fetchTasks() {
    loading.value = true
    const q = new URLSearchParams()
    if (search.value)         q.set('search',   search.value)
    if (statusFilter.value)   q.set('status',   statusFilter.value)
    if (priorityFilter.value) q.set('priority', priorityFilter.value)
    try {
      const res  = await fetch(`/api/tasks?${q}`)
      tasks.value = await res.json()
    } catch {
      tasks.value = []
    } finally {
      loading.value = false
    }
  }

  function openAdd() {
    editingId.value = null
    form.value = emptyForm()
    showModal.value = true
  }

  function openEdit(t) {
    editingId.value = t.id
    form.value = {
      title:       t.title,
      status:      t.status,
      priority:    t.priority,
      assignee:    t.assignee    || '',
      due_date:    t.due_date    ? t.due_date.slice(0, 10) : '',
      description: t.description || '',
    }
    showModal.value = true
  }

  async function saveTask(formData) {
    const url    = editingId.value ? `/api/tasks/${editingId.value}` : '/api/tasks'
    const method = editingId.value ? 'PUT' : 'POST'
    await fetch(url, {
      method,
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ ...formData, due_date: formData.due_date || null })
    })
    showModal.value = false
    fetchTasks()
  }

  async function deleteTask(id) {
    await fetch(`/api/tasks/${id}`, { method: 'DELETE' })
    confirmDelete.value = null
    fetchTasks()
  }
t.due_date.slice(0, 10)ตัดเวลาออก เหลือแค่ YYYY-MM-DD — input[type=date] ต้องการ format นี้
due_date || nullส่ง null ถ้า due_date ว่าง — ไม่ส่ง empty string ไปให้ DB

ส่วนที่ 5 — Watchers (3 ตัว) + Return:

useTasks.js — Watchers + Return
  // ── Watchers ──────────────────────────────────────────────────────
  let debounceTimer = null
  watch(search, () => {
    clearTimeout(debounceTimer)
    debounceTimer = setTimeout(fetchTasks, 280)   // รอ 280ms หลังพิมพ์หยุด
  })

  watch(statusFilter,   fetchTasks)   // fetch ทันทีเมื่อเลือก status
  watch(priorityFilter, fetchTasks)   // fetch ทันทีเมื่อเลือก priority

  return {
    tasks, loading, search, statusFilter, priorityFilter,
    showModal, editingId, confirmDelete, form,
    filtered, stats,
    fetchTasks, openAdd, openEdit, saveTask, deleteTask,
  }
}
watch(search, debounce 280ms)พิมพ์ทีละ character → ไม่ fetch ทุก keystroke — รอ 280ms หลังหยุดพิมพ์แล้วค่อย fetch
watch(statusFilter, fetchTasks)เลือก dropdown → fetch ทันที ไม่ต้อง debounce เพราะ user เลือกครั้งเดียว
watch(priorityFilter, fetchTasks)เหมือน statusFilter — เพิ่มจาก Week 3 ที่มีแค่ catFilter
🔄
Computed filtered vs fetchTasks filtered computed กรอง tasks ใน memory — ไม่ส่ง request ไป server
fetchTasks ส่ง query parameters ไป API — server filter ใน PostgreSQL
ทั้งคู่ทำงานพร้อมกัน: API filter ลด payload, computed filter ทำให้ UI responsive

defineProps({ task: { type: Object, required: true } })
defineEmits(['edit', 'delete'])

// Status: CSS class
function statusClass(s)   {
  return { 'todo': 's-todo', 'in-progress': 's-prog', 'done': 's-done' }[s] || 's-todo'
}
// Status: ป้ายภาษาไทย
function statusLabel(s)   {
  return { 'todo': 'รอดำเนินการ', 'in-progress': 'กำลังทำ', 'done': 'เสร็จสิ้น' }[s] || s
}
// Priority: CSS class สำหรับ badge
function priorityClass(p) {
  return { 'low': 'p-low', 'medium': 'p-med', 'high': 'p-high' }[p] || 'p-med'
}
// Priority: ป้ายภาษาไทย
function priorityLabel(p) {
  return { 'low': 'ต่ำ', 'medium': 'ปานกลาง', 'high': 'สูง' }[p] || p
}
// Priority: สีขอบ card ด้านซ้าย
function priorityBorderClass(p) {
  return { 'low': 'border-low', 'medium': 'border-med', 'high': 'border-high' }[p] || 'border-med'
}
// Helper overdue (เหมือนใน composable)
function isOverdue(t) {
  return t.due_date && new Date(t.due_date) < new Date() && t.status !== 'done'
}
// Format วันที่เป็น DD MMM YYYY ภาษาไทย
function formatDate(d) {
  if (!d) return ''
  return new Date(d).toLocaleDateString('th-TH', {
    year: 'numeric', month: 'short', day: 'numeric'
  })
}

App.vue — script setup
import { onMounted } from 'vue'
import { useTasks }     from './composables/useTasks.js'
import AppHeader        from './components/AppHeader.vue'
import StatsGrid        from './components/StatsGrid.vue'
import TaskToolbar      from './components/TaskToolbar.vue'
import TaskGrid         from './components/TaskGrid.vue'
import TaskModal        from './components/TaskModal.vue'
import DeleteConfirm    from './components/DeleteConfirm.vue'

// destructure ทุกอย่างจาก useTasks composable
const {
  tasks, loading, search, statusFilter, priorityFilter,
  showModal, editingId, confirmDelete, form,
  filtered, stats,
  fetchTasks, openAdd, openEdit, saveTask, deleteTask,
} = useTasks()

onMounted(fetchTasks)   // fetch ครั้งแรกเมื่อ App mount
App.vue — template ทั้งหมด
<template>
  <div>
    <AppHeader @add="openAdd" />

    <main class="main">
      <StatsGrid :stats="stats" />

      <TaskToolbar
        v-model:search="search"
        v-model:statusFilter="statusFilter"
        v-model:priorityFilter="priorityFilter"
        :loading="loading"
        :filtered-count="filtered.length"
        :total-count="tasks.length"
      />

      <TaskGrid
        :tasks="filtered"
        :loading="loading"
        :is-filtered="!!(search || statusFilter || priorityFilter)"
        @edit="openEdit"
        @request-delete="t => (confirmDelete = t)"
      />
    </main>

    <!-- v-if: mount ใหม่ทุกครั้ง → localForm ใน Modal reset เสมอ -->
    <TaskModal
      v-if="showModal"
      :editing-id="editingId"
      :initial-form="form"
      @save="saveTask"
      @close="showModal = false"
    />
    <DeleteConfirm
      v-if="confirmDelete"
      :task="confirmDelete"
      @confirm="deleteTask(confirmDelete.id)"
      @cancel="confirmDelete = null"
    />
  </div>
</template>
v-model:search="search"TaskToolbar เปลี่ยน search → emit update:search → search ref เปลี่ยน → watch → debounce → fetchTasks
v-model:statusFilterเหมือน search แต่ใช้ watch(statusFilter) ที่ไม่มี debounce — fetch ทันที
:tasks="filtered"ส่ง filtered (ไม่ใช่ tasks raw) ให้ TaskGrid — grid แสดงเฉพาะที่ผ่าน filter
@request-delete="t => confirmDelete = t"set confirmDelete → DeleteConfirm แสดงขึ้น (v-if="confirmDelete")
v-if="showModal"destroy + re-mount TaskModal ทุกครั้ง → localForm = reactive({ ...initialForm }) เสมอ
