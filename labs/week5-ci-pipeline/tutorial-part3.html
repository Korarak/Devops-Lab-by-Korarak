Jest + Supertest

Test ที่ง่ายที่สุดคือ GET /health เพราะ endpoint นี้ ไม่ query DB เลย — ไม่ต้อง mock อะไรทั้งนั้น

เปิดไฟล์: backend/tests/health.test.js
backend/tests/health.test.js — เนื้อหาทั้งหมด
const request = require('supertest');
const app = require('../index');

describe('GET /health', () => {
  test('returns 200 with status ok', async () => {
    const res = await request(app).get('/health');
    expect(res.statusCode).toBe(200);
    expect(res.body).toHaveProperty('status', 'ok');
  });

  test('returns version field', async () => {
    const res = await request(app).get('/health');
    expect(res.body).toHaveProperty('version');
    expect(typeof res.body.version).toBe('string');
  });

  test('returns uptime as a number', async () => {
    const res = await request(app).get('/health');
    expect(res.body).toHaveProperty('uptime');
    expect(typeof res.body.uptime).toBe('number');
    expect(res.body.uptime).toBeGreaterThanOrEqual(0);
  });

  test('returns service containing TaskFlow', async () => {
    const res = await request(app).get('/health');
    expect(res.body.service).toContain('TaskFlow');
  });
});

อธิบาย API ของ Jest + Supertest:

request(app) supertest สร้าง HTTP server จาก express app โดยตรง — ไม่ต้อง app.listen() ก่อน, Jest จัดการ lifecycle ให้
describe() จัดกลุ่ม test ที่เกี่ยวข้องกัน — ช่วยให้ output อ่านง่ายและ test ถูก organise
test() test case เดียว — แต่ละ test ควร test สิ่งเดียวเป็นหลัก (single responsibility)
expect().toBe() assert ว่าค่า ตรงกันเป๊ะ (strict equality) — ใช้กับ primitive: number, string, boolean
expect().toHaveProperty() assert ว่า object มี property นั้น — อาจส่ง argument ที่สองเพื่อตรวจค่าด้วย
expect().toContain() assert ว่า string มี substring นั้นอยู่ — เหมาะเมื่อไม่ต้องการตรงทั้งหมด
toBeGreaterThanOrEqual(0) assert ว่าตัวเลข ≥ 0 — ใช้ตรวจ uptime ที่ไม่รู้ค่าแน่นอน

tasks.test.js — Mock Database
jest.mock()
⚠️
ทำไมต้อง Mock Database? ถ้า ไม่ mock → ต้องมี PostgreSQL จริงๆ รันอยู่ใน CI → ซับซ้อน, ช้า, อาจล้มเหลวเพราะ network latency หรือ DB ไม่พร้อม — ทำให้ test ไม่ stable (flaky)
เปิดไฟล์: backend/tests/tasks.test.js
backend/tests/tasks.test.js — เนื้อหาทั้งหมด
const request = require('supertest');

jest.mock('../db', () => ({
  query: jest.fn(),
}));
const db  = require('../db');
const app = require('../index');

describe('GET /api/tasks', () => {
  beforeEach(() => jest.clearAllMocks());

  test('returns 200 with array of tasks', async () => {
    db.query.mockResolvedValueOnce({
      rows: [
        { id: 1, title: 'เขียน Unit Test', status: 'todo', priority: 'high', assignee: 'ปิยะ' },
        { id: 2, title: 'สร้าง CI Pipeline', status: 'in-progress', priority: 'high', assignee: 'สมชาย' },
      ],
    });
    const res = await request(app).get('/api/tasks');
    expect(res.statusCode).toBe(200);
    expect(Array.isArray(res.body)).toBe(true);
    expect(res.body).toHaveLength(2);
  });

  test('returns empty array when no tasks', async () => {
    db.query.mockResolvedValueOnce({ rows: [] });
    const res = await request(app).get('/api/tasks');
    expect(res.statusCode).toBe(200);
    expect(res.body).toEqual([]);
  });
});

describe('GET /api/tasks/:id', () => {
  beforeEach(() => jest.clearAllMocks());

  test('returns 200 with single task', async () => {
    db.query.mockResolvedValueOnce({
      rows: [{ id: 1, title: 'เขียน Unit Test', status: 'todo', priority: 'high' }],
    });
    const res = await request(app).get('/api/tasks/1');
    expect(res.statusCode).toBe(200);
    expect(res.body).toHaveProperty('id', 1);
  });

  test('returns 404 when task not found', async () => {
    db.query.mockResolvedValueOnce({ rows: [] });
    const res = await request(app).get('/api/tasks/999');
    expect(res.statusCode).toBe(404);
  });
});

describe('POST /api/tasks', () => {
  beforeEach(() => jest.clearAllMocks());

  test('creates task and returns 201', async () => {
    db.query.mockResolvedValueOnce({
      rows: [{ id: 3, title: 'งานใหม่', status: 'todo', priority: 'medium', assignee: 'ปิยะ' }],
    });
    const res = await request(app)
      .post('/api/tasks')
      .send({ title: 'งานใหม่', status: 'todo', priority: 'medium', assignee: 'ปิยะ' });
    expect(res.statusCode).toBe(201);
    expect(res.body).toHaveProperty('id');
    expect(res.body.title).toBe('งานใหม่');
  });
});

describe('DELETE /api/tasks/:id', () => {
  beforeEach(() => jest.clearAllMocks());

  test('returns 200 when deleted', async () => {
    db.query.mockResolvedValueOnce({
      rows: [{ id: 1, title: 'งานทดสอบ' }],
    });
    const res = await request(app).delete('/api/tasks/1');
    expect(res.statusCode).toBe(200);
  });

  test('returns 404 when task not found', async () => {
    db.query.mockResolvedValueOnce({ rows: [] });
    const res = await request(app).delete('/api/tasks/999');
    expect(res.statusCode).toBe(404);
  });
});

อธิบาย Mock Pattern ที่สำคัญ:

jest.mock('../db', ...) ต้องมาก่อน require('../index') — Jest intercept ก่อน module ถูก load เข้า memory เพื่อให้ app ได้รับ mock แทน DB จริง
query: jest.fn() สร้าง mock function — จับ calls ได้ทั้งหมด, กำหนดค่า return ได้, reset ได้
mockResolvedValueOnce({rows:[...]}) กำหนดว่า db.query() จะ resolve ด้วยค่านี้ ครั้งเดียว (once) — การเรียกครั้งถัดไปจะ return undefined
beforeEach(() => jest.clearAllMocks()) reset mock state ก่อนแต่ละ test — ป้องกัน mock ที่ set ใน test ก่อนหน้า "ค้าง" มาส่งผลกับ test ถัดไป
mockResolvedValueOnce vs mockResolvedValue Once = ใช้ได้รอบเดียวแล้วหมด, ไม่มี Once = ทุกครั้งที่เรียก query ได้ค่าเดิม
🔍
ลำดับที่สำคัญมาก: jest.mock → require db → require app Jest hoist jest.mock() ขึ้นบนสุดของ file อัตโนมัติ ทำให้ทุก module ที่ require หลังจากนั้นได้รับ mock version

const request = require('supertest');

jest.mock('../db', () => ({ query: jest.fn() }));
const db  = require('../db');
const app = require('../index');

describe('POST /api/tasks — validation', () => {
  beforeEach(() => jest.clearAllMocks());

  test('returns 400 when title is missing', async () => {
    const res = await request(app)
      .post('/api/tasks')
      .send({ status: 'todo', priority: 'medium' });
    expect(res.statusCode).toBe(400);
    expect(res.body).toHaveProperty('error');
  });

  test('returns 400 when status is invalid', async () => {
    const res = await request(app)
      .post('/api/tasks')
      .send({ title: 'งานทดสอบ', status: 'invalid-status' });
    expect(res.statusCode).toBe(400);
    expect(res.body).toHaveProperty('error');
  });

  test('returns 400 when priority is invalid', async () => {
    const res = await request(app)
      .post('/api/tasks')
      .send({ title: 'งานทดสอบ', priority: 'urgent' });
    expect(res.statusCode).toBe(400);
    expect(res.body).toHaveProperty('error');
  });

  test('returns 201 when only title provided', async () => {
    db.query.mockResolvedValueOnce({
      rows: [{ id: 5, title: 'งานใหม่', status: 'todo', priority: 'medium' }],
    });
    const res = await request(app)
      .post('/api/tasks')
      .send({ title: 'งานใหม่' });

  });
});

ตัวอย่าง output coverage ที่จะเห็น:

terminal output — coverage summary

File                | % Stmts | % Branch | % Funcs | % Lines |

index.js            |   72.31 |    64.29 |   87.50 |   72.31 |

ความหมายของแต่ละ metric:

% Lines กี่ % ของบรรทัดทั้งหมดถูก execute ระหว่าง test — metric หลักที่ใช้วัด coverage
% Branch กี่ % ของ if/else ถูก test ทั้งสองทาง — ต่ำกว่า Lines เสมอเพราะต้องครอบทุก condition
% Functions กี่ % ของ function ถูกเรียกอย่างน้อยครั้งหนึ่ง
% Statements กี่ % ของ statement (expression, declaration) ถูก execute — ใกล้เคียง Lines มาก
📊
HTML Coverage Report หลังรัน npm test -- --coverage จะมีไฟล์ที่ backend/coverage/lcov-report/index.html เปิดใน browser ได้ — เห็นบรรทัดไหนถูก/ไม่ถูก execute สีเขียว/แดงชัดเจน
⚠️
ถ้า coverage ต่ำกว่า threshold → CI fail! jest.config.js กำหนด threshold ≥ 60% lines, ≥ 60% functions, ≥ 50% branches ถ้า coverage ต่ำกว่านี้ npm test -- --coverage จะ exit ด้วย error code

GitHub Actions CI Workflow — ci.yml
GitHub Actions

ไฟล์ .github/workflows/ci.yml คือหัวใจของ CI Pipeline — GitHub Actions จะอ่านไฟล์นี้และรัน 3 jobs อัตโนมัติทุกครั้งที่ push

เปิดไฟล์: .github/workflows/ci.yml
.github/workflows/ci.yml — เนื้อหาทั้งหมด
name: CI — TaskFlow

on:
  push:
    branches:
      - main
      - develop
      - 'feature/**'
      - 'hotfix/**'
    paths:
      - 'backend/**'
      - '.github/workflows/ci.yml'
  pull_request:
    branches:
      - main
      - develop
    types: [opened, synchronize, reopened]

jobs:
  # ─── JOB 1: LINT ────────────────────────────────────────────
  lint:
    name: 🔍 Lint
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: backend

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Run ESLint
        run: npm run lint

  # ─── JOB 2: TEST ────────────────────────────────────────────
  test:
    name: 🧪 Test & Coverage
    runs-on: ubuntu-latest
    needs: lint
    defaults:
      run:
        working-directory: backend

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Run tests with coverage
        run: npm test
        env:
          NODE_ENV: test

      - name: Upload coverage report
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: coverage-report
          path: backend/coverage/
          retention-days: 7

  # ─── JOB 3: BUILD ───────────────────────────────────────────
  build:
    name: 🐳 Docker Build
    runs-on: ubuntu-latest
    needs: test

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Build Docker image (smoke test)
        run: |
          docker build -t taskflow-backend:${{ github.sha }} ./backend
          echo "✅ Build successful: taskflow-backend:${{ github.sha }}"

อธิบาย configuration ที่สำคัญ:

on.push.paths CI ทำงานเฉพาะเมื่อไฟล์ใน backend/ หรือ ci.yml เปลี่ยน — ไม่ waste time ถ้าแก้แค่ frontend หรือ docs
needs: lint test job รอให้ lint job ผ่านก่อน — ถ้า lint fail, test ไม่รัน ประหยัด CI minutes
needs: test build job รอให้ test job ผ่านก่อน — Docker image จะ build เฉพาะเมื่อ tests ผ่านแล้วเท่านั้น
defaults.run.working-directory: backend ทุก step ใน job นั้น cd เข้า backend อัตโนมัติ — ไม่ต้องพิมพ์ cd backend ทุก step
cache: 'npm' + cache-dependency-path cache node_modules ระหว่าง runs โดยใช้ package-lock.json เป็น cache key — เร็วขึ้น ~70%
npm ci vs npm install npm ci = clean install ตาม package-lock.json เป๊ะ, ไม่แก้ lock file — reproducible build ใน CI
if: always() upload coverage artifact แม้ test fail — ทำให้ดู coverage report ได้เสมอเพื่อ debug
NODE_ENV: test บอก app ว่าอยู่ใน test environment — ป้องกัน app พยายาม connect DB จริงระหว่าง test

Pipeline flow ภาพรวม:

Push to feature/* branch
↓ Job 1: 🔍 Lint
ESLint check — ตรวจ code style + rules ↓ (needs: lint) Job 2: 🧪 Test & Coverage
15 tests + coverage report + upload artifact ↓ (needs: test) Job 3: 🐳 Docker Build
docker build smoke test — ตรวจ Dockerfile ใช้ได้

ESLint Configuration — .eslintrc.json:

เปิดไฟล์: backend/.eslintrc.json
backend/.eslintrc.json
{
  "env": { "node": true, "es2021": true, "jest": true },
  "extends": "eslint:recommended",
  "parserOptions": { "ecmaVersion": 2021 },
  "rules": {
    "no-unused-vars": "warn",
    "no-console": "off",
    "semi": ["error", "always"],
    "quotes": ["warn", "single"],
    "eqeqeq": ["error", "always"],
    "no-var": "error",
    "prefer-const": "warn"
  }
}

อธิบาย ESLint rules ที่สำคัญ:

"semi": ["error", "always"] ต้องมี semicolon ทุก statement — CI fail ถ้าขาด (error = breaking)
"eqeqeq": ["error", "always"] ต้องใช้ === ไม่ใช่ == — ป้องกัน type coercion bugs
"no-var": "error" ห้ามใช้ var — ต้องใช้ const หรือ let เท่านั้น
"jest": true บอก ESLint รู้จัก jest globals: describe, test, expect, beforeEach — ไม่ error ว่าไม่ได้ declare
"no-console": "off" อนุญาต console.log — backend มักต้องการ log ข้อมูล
🐛 Debug Challenge — อ่าน CI Log แล้วหาปัญหา

Scenario: มี broken test ที่ CI fail — test expects status 200 แต่ได้ 500 เพราะ mock ไม่ได้ถูกตั้งค่า

วิธีแก้ step by step:

  1. ดู error message ใน Actions tab — หา "Expected: 200, Received: 500" หรือ stack trace
  2. เปิด test file ที่ fail — ดูว่า mockResolvedValueOnce ถูก set ก่อน request หรือไม่
  3. ตรวจว่า jest.mock('../db') อยู่ก่อน require('../index') ในไฟล์
  4. ตรวจ beforeEach(() => jest.clearAllMocks()) — ถ้าขาดอาจ mock ค้างจาก test ก่อน
  5. แก้ไข → git push → ดู CI run ใหม่ใน Actions tab
