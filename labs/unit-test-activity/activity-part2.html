6
AAA Pattern — โครงสร้าง Test ที่ดี
⏱ 8 นาที

AAA Pattern (Arrange-Act-Assert) คือโครงสร้างมาตรฐานของ Unit Test ที่ช่วยให้โค้ด test อ่านง่ายและเป็นระเบียบ ทุก Test ควรแบ่งเป็น 3 ส่วนนี้

Arrange
จัดเตรียม
เตรียมข้อมูล, ค่า input, และสิ่งที่จำเป็นก่อน test
Act
กระทำ
เรียกฟังก์ชัน/method ที่ต้องการทดสอบ
Assert
ตรวจสอบ
ตรวจสอบว่าผลลัพธ์ตรงกับที่คาดหวังหรือไม่

ตัวอย่าง Test ที่ไม่ได้ใช้ AAA (อ่านยาก)

❌ ไม่แนะนำ
test('average test', () => {
  expect(average([80, 90, 70])).toBe(80); // ยัดทุกอย่างในบรรทัดเดียว อ่านยาก
});

ตัวอย่าง Test ที่ใช้ AAA Pattern (อ่านง่าย)

✅ แนะนำ — tests/calculator.test.js
test('average: คะแนน 3 วิชาต้องได้เฉลี่ยถูกต้อง', () => {
  // Arrange — เตรียมข้อมูล input
  const scores = [80, 90, 70];
  const expectedAverage = 80;

  // Act — เรียกฟังก์ชันที่ต้องการทดสอบ
  const result = average(scores);

  // Assert — ตรวจสอบผลลัพธ์
  expect(result).toBe(expectedAverage);
});

test('getGrade: คะแนน 85 ต้องได้เกรด A', () => {
  // Arrange
  const score = 85;

  // Act
  const grade = getGrade(score);

  // Assert
  expect(grade).toBe('A');
});
✅
ข้อดีของ AAA Pattern
  • เห็นชัดว่าแต่ละส่วนทำอะไร — คนอื่นอ่านแล้วเข้าใจทันที
  • เมื่อ test fail จะรู้ว่าปัญหาอยู่ที่ Act หรือ Assert
  • ง่ายต่อการ maintain และแก้ไขในอนาคต
💡
เคล็ดลับถ้า "Act" ส่วนของคุณมีมากกว่า 1 บรรทัด อาจเป็นสัญญาณว่า function นั้นทำงานมากเกินไป (ควร refactor)

describe() + it() — จัดกลุ่ม Test
⏱ 8 นาที

describe() ใช้สำหรับ จัดกลุ่ม tests ที่เกี่ยวข้องกัน ส่วน it() เป็นคำพ้องความหมายกับ test() ที่ทำให้อ่านเป็นประโยคได้

ก่อนใช้ describe (ยุ่งเหยิง)

❌ ทุก test ปะปนกัน
test('add: 2+3 = 5', ...);
test('average: [80,90,70] = 80', ...);
test('add: 0+0 = 0', ...);
test('getGrade: 85 = A', ...);
test('average: [] = 0', ...);

หลังใช้ describe (เป็นระเบียบ)

✅ tests/calculator.test.js — เวอร์ชันปรับปรุง
const { add, subtract, multiply, divide, average, getGrade } = require('../src/utils/calculator');

// describe() จัดกลุ่มตามฟังก์ชัน
describe('add()', () => {
  it('บวกเลขบวกสองตัวได้ถูกต้อง', () => {
    expect(add(2, 3)).toBe(5);
  });

  it('บวกด้วย 0 ได้ถูกต้อง', () => {
    expect(add(5, 0)).toBe(5);
  });

  it('บวกเลขลบได้ถูกต้อง', () => {
    expect(add(-3, 7)).toBe(4);
  });
});

describe('average()', () => {
  it('คำนวณเฉลี่ยจาก array ปกติได้ถูกต้อง', () => {
    expect(average([80, 90, 70])).toBe(80);
  });

  it('คืนค่า 0 เมื่อ array ว่าง', () => {
    expect(average([])).toBe(0);
  });

  it('คำนวณเฉลี่ยจากตัวเลขเดียวได้ถูกต้อง', () => {
    expect(average([100])).toBe(100);
  });
});

describe('getGrade()', () => {
  it('คะแนน 80-100 ได้เกรด A', () => {
    expect(getGrade(80)).toBe('A');
    expect(getGrade(100)).toBe('A');
  });

  it('คะแนน 70-79 ได้เกรด B', () => {
    expect(getGrade(70)).toBe('B');
    expect(getGrade(79)).toBe('B');
  });

  it('คะแนน 60-69 ได้เกรด C', () => {
    expect(getGrade(60)).toBe('C');
  });

  it('คะแนน 50-59 ได้เกรด D', () => {
    expect(getGrade(50)).toBe('D');
  });

  it('คะแนนต่ำกว่า 50 ได้เกรด F', () => {
    expect(getGrade(49)).toBe('F');
    expect(getGrade(0)).toBe('F');
  });
});

ผลลัพธ์ที่ได้ — เป็นระเบียบชัดเจน

📌
it() vs test()ทั้งสองเหมือนกันทุกอย่าง — ใช้ it() เมื่อต้องการให้ชื่อ test อ่านเป็นประโยค เช่น "it should return 5" หรือ "it คืนค่า 0 เมื่อ array ว่าง"

Matchers สำคัญ — ตรวจสอบผลลัพธ์ได้ทุกรูปแบบ
⏱ 10 นาที

Matchers คือฟังก์ชันที่ต่อท้าย expect() เพื่อบอกว่าต้องการตรวจสอบอะไร Jest มี matchers หลายสิบตัว เราจะเรียน 12 ตัวที่ใช้บ่อยที่สุด

Matcherใช้เมื่อตัวอย่าง
.toBe(val)ตรวจค่าแบบ strict (===)expect(1+1).toBe(2)
.toEqual(obj)ตรวจ object/array (เปรียบค่า)expect([1,2]).toEqual([1,2])
.toBeNull()ตรวจว่าเป็น nullexpect(result).toBeNull()
.toBeUndefined()ตรวจว่าเป็น undefinedexpect(val).toBeUndefined()
.toBeTruthy()ตรวจค่า truthy (ไม่ว่าง)expect('hello').toBeTruthy()
.toBeFalsy()ตรวจค่า falsy (ว่าง/false/0)expect(0).toBeFalsy()
.toContain(item)ตรวจว่า array/string มี itemexpect([1,2,3]).toContain(2)
.toHaveLength(n)ตรวจความยาว array/stringexpect([1,2,3]).toHaveLength(3)
.toBeGreaterThan(n)ตรวจว่ามากกว่าexpect(5).toBeGreaterThan(3)
.toBeLessThan(n)ตรวจว่าน้อยกว่าexpect(2).toBeLessThan(5)
.toMatch(regex)ตรวจ string ด้วย regexexpect('hello').toMatch(/ell/)
.toThrow()ตรวจว่า function throw errorexpect(() => fn()).toThrow()

ตัวอย่างการใช้งาน Matchers กับ ScoreCalc

tests/matchers-demo.test.js
const { add, average, getGrade } = require('../src/utils/calculator');

describe('ตัวอย่าง Matchers ต่างๆ', () => {

  // toBe — ตรวจค่า primitive (number, string, boolean)
  it('toBe: ตรวจค่าตัวเลข', () => {
    expect(add(2, 3)).toBe(5);
  });

  // toEqual — ตรวจ object หรือ array (ห้ามใช้ toBe กับ object!)
  it('toEqual: ตรวจ array', () => {
    const grades = ['A', 'B', 'C'].map(g => g);
    expect(grades).toEqual(['A', 'B', 'C']);
  });

  // toBeGreaterThanOrEqual — ตรวจ range
  it('toBeGreaterThanOrEqual: ตรวจ threshold', () => {
    const avg = average([75, 80, 85]);
    expect(avg).toBeGreaterThanOrEqual(70); // ต้องได้คะแนนเฉลี่ยอย่างน้อย 70
  });

  // toBeTruthy — ตรวจว่ามีค่า
  it('toBeTruthy: ตรวจว่าได้เกรด', () => {
    const grade = getGrade(75);
    expect(grade).toBeTruthy(); // ต้องไม่ใช่ null, undefined, '', 0
  });

  // toHaveLength — ตรวจความยาว
  it('toHaveLength: เกรดต้องเป็น string 1 ตัวอักษร', () => {
    expect(getGrade(85)).toHaveLength(1);
  });

  // toContain — ตรวจว่า array มีค่า
  it('toContain: เกรดที่เป็นไปได้', () => {
    const validGrades = ['A', 'B', 'C', 'D', 'F'];
    expect(validGrades).toContain(getGrade(85));
  });

});
⚠️
ข้อควรระวัง: toBe vs toEqual
ตัวอย่างความแตกต่าง
// ❌ จะ FAIL — object ต่างตัวกัน แม้ค่าเดียวกัน
expect({name: 'test'}).toBe({name: 'test'});

// ✅ จะ PASS — toEqual เปรียบค่าภายใน
expect({name: 'test'}).toEqual({name: 'test'});

Test Edge Cases + toThrow() — กรณีพิเศษ
⏱ 10 นาที

Edge Cases คือกรณีพิเศษที่อยู่ที่ขอบเขตของ input เช่น ค่า 0, ค่าลบ, string ว่าง, null, undefined หรือค่าที่ใหญ่มากๆ — มักเป็นแหล่งที่ bug ซ่อนอยู่

1. เพิ่ม Edge Cases ให้ getGrade()

ลองคิดว่า function ควรทำงานอย่างไรเมื่อรับค่าที่ไม่ปกติ ก่อนอื่นให้อัปเดตไฟล์โค้ดก่อน:

src/utils/calculator.js — เพิ่ม validation
function divide(a, b) {
  if (b === 0) {
    throw new Error('ไม่สามารถหารด้วย 0 ได้');
  }
  return a / b;
}

function average(scores) {
  if (!Array.isArray(scores)) {
    throw new TypeError('scores ต้องเป็น Array');
  }
  if (scores.length === 0) return 0;
  const sum = scores.reduce((acc, val) => acc + val, 0);
  return sum / scores.length;
}

function getGrade(score) {
  if (typeof score !== 'number' || score < 0 || score > 100) {
    throw new RangeError('คะแนนต้องเป็นตัวเลข 0-100');
  }
  if (score >= 80) return 'A';
  if (score >= 70) return 'B';
  if (score >= 60) return 'C';
  if (score >= 50) return 'D';
  return 'F';
}

2. เขียน Test สำหรับ Edge Cases และ Error

tests/edge-cases.test.js
const { divide, average, getGrade } = require('../src/utils/calculator');

describe('Edge Cases — กรณีพิเศษ', () => {

  describe('divide()', () => {
    it('หารปกติได้ถูกต้อง', () => {
      expect(divide(10, 2)).toBe(5);
    });

    // toThrow() — ตรวจว่า function 던져 error
    it('หารด้วย 0 ต้อง throw Error', () => {
      // ต้องห่อด้วย arrow function!
      expect(() => divide(10, 0)).toThrow('ไม่สามารถหารด้วย 0 ได้');
    });

    it('หารด้วย 0 ต้อง throw instance ของ Error', () => {
      expect(() => divide(5, 0)).toThrow(Error);
    });
  });

  describe('average() — Edge Cases', () => {
    it('คำนวณจากตัวเลขเดียวได้', () => {
      expect(average([85])).toBe(85);
    });

    it('คำนวณจากคะแนนทศนิยมได้', () => {
      expect(average([1, 2])).toBe(1.5);
    });

    it('array ว่างต้องคืนค่า 0', () => {
      expect(average([])).toBe(0);
    });

    it('รับ input ที่ไม่ใช่ Array ต้อง throw TypeError', () => {
      expect(() => average('ไม่ใช่ array')).toThrow(TypeError);
    });
  });

  describe('getGrade() — Boundary Values', () => {
    it('คะแนน 80 อยู่ที่ขอบล่างของเกรด A', () => {
      expect(getGrade(80)).toBe('A');
    });

    it('คะแนน 79 อยู่ที่ขอบบนของเกรด B', () => {
      expect(getGrade(79)).toBe('B');
    });

    it('คะแนน 0 ต้องได้เกรด F', () => {
      expect(getGrade(0)).toBe('F');
    });

    it('คะแนน 100 ต้องได้เกรด A', () => {
      expect(getGrade(100)).toBe('A');
    });

    it('คะแนน -1 ต้อง throw RangeError', () => {
      expect(() => getGrade(-1)).toThrow(RangeError);
    });

    it('คะแนน 101 ต้อง throw RangeError', () => {
      expect(() => getGrade(101)).toThrow(RangeError);
    });

    it('input ที่ไม่ใช่ตัวเลข ต้อง throw RangeError', () => {
      expect(() => getGrade('A')).toThrow(RangeError);
      expect(() => getGrade(null)).toThrow(RangeError);
    });
  });
});
🎯
กลยุทธ์การเลือก Edge Cases
  • Boundary Values: ค่าที่ขอบบน/ล่าง เช่น 79, 80, 81
  • Empty/Null: array ว่าง, string ว่าง, null, undefined
  • ค่าลบ/ค่าผิดประเภท: -1, NaN, string แทนตัวเลข
  • ค่าขนาดใหญ่มาก: Infinity, Number.MAX_SAFE_INTEGER

beforeEach / afterEach — จัดการ State
⏱ 9 นาที

beforeEach และ afterEach คือ lifecycle hooks ที่รันโค้ด ก่อน และ หลัง แต่ละ test — มีประโยชน์มากเมื่อ tests หลายตัวต้องการข้อมูลชุดเดียวกัน

ปัญหาเมื่อไม่ใช้ beforeEach (โค้ดซ้ำ)

❌ โค้ดซ้ำในทุก test
describe('ScoreBook', () => {
  it('เพิ่มคะแนนได้', () => {
    const book = new ScoreBook(); // ซ้ำทุก test
    book.add('สมชาย', 85);
    expect(book.size()).toBe(1);
  });

  it('ลบคะแนนได้', () => {
    const book = new ScoreBook(); // ซ้ำทุก test
    book.add('สมชาย', 85);
    book.remove('สมชาย');
    expect(book.size()).toBe(0);
  });
});

เพิ่มไฟล์ ScoreBook ก่อน

src/utils/scorebook.js
class ScoreBook {
  constructor() {
    this.scores = {};
  }

  add(name, score) {
    this.scores[name] = score;
  }

  remove(name) {
    delete this.scores[name];
  }

  get(name) {
    return this.scores[name];
  }

  size() {
    return Object.keys(this.scores).length;
  }

  getAll() {
    return { ...this.scores };
  }
}

module.exports = ScoreBook;

ใช้ beforeEach และ afterEach

✅ tests/scorebook.test.js
const ScoreBook = require('../src/utils/scorebook');
const { getGrade } = require('../src/utils/calculator');

describe('ScoreBook', () => {
  let book; // ประกาศไว้ด้านนอก

  // beforeEach รันก่อนทุก test — สร้าง instance ใหม่ทุกครั้ง
  beforeEach(() => {
    book = new ScoreBook();
    book.add('สมชาย', 85);
    book.add('สมหญิง', 72);
    console.log('เตรียม ScoreBook แล้ว');
  });

  // afterEach รันหลังทุก test — cleanup
  afterEach(() => {
    book = null; // ล้างค่า เพื่อให้แต่ละ test เป็นอิสระ
  });

  it('ข้อมูลเริ่มต้นต้องมี 2 คน', () => {
    expect(book.size()).toBe(2);
  });

  it('เพิ่มนักเรียนใหม่ได้', () => {
    book.add('สมศรี', 90);
    expect(book.size()).toBe(3);
    expect(book.get('สมศรี')).toBe(90);
  });

  it('ลบนักเรียนได้', () => {
    book.remove('สมชาย');
    expect(book.size()).toBe(1);
  });

  it('ดึงคะแนนของนักเรียนได้', () => {
    expect(book.get('สมชาย')).toBe(85);
    expect(book.get('สมหญิง')).toBe(72);
  });

  it('ดึงเกรดของนักเรียนจากคะแนนใน book ได้', () => {
    const score = book.get('สมชาย');
    expect(getGrade(score)).toBe('A'); // 85 = A
  });
});
📌
Lifecycle Hooks ทั้งหมด
  • beforeAll(() => {...}) — รัน 1 ครั้ง ก่อน describe block ทั้งหมด
  • beforeEach(() => {...}) — รันก่อนทุก test ใน describe
  • afterEach(() => {...}) — รันหลังทุก test ใน describe
  • afterAll(() => {...}) — รัน 1 ครั้ง หลัง describe block ทั้งหมด
