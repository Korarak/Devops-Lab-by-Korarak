ฟังก์ชันที่ต้องออกแบบ Test

ฟังก์ชัน isPassScore()
/**
 * ตรวจสอบว่านักเรียนผ่านเกณฑ์หรือไม่
 * @param {number} score - คะแนนที่ได้ (0-100)
 * @param {number} passThreshold - เกณฑ์ผ่าน (ค่า default = 50)
 * @returns {boolean} true ถ้าผ่าน, false ถ้าไม่ผ่าน
 * @throws {Error} ถ้า score ไม่ใช่ตัวเลขหรืออยู่นอกช่วง 0-100
 */
function isPassScore(score, passThreshold = 50) {
  if (typeof score !== 'number' || score < 0 || score > 100) {
    throw new Error('Invalid score');
  }
  return score >= passThreshold;
}

describe('isPassScore()', () => {
  // 1. Normal pass — default threshold 50
  it('should return true when score >= 50 (default)', () => {
    expect(isPassScore(75)).toBe(true);
  });

  // 2. Boundary — ขอบล่างของการผ่าน
  it('should return true when score equals threshold exactly (50)', () => {
    expect(isPassScore(50)).toBe(true);
  });

  // 3. Boundary — ขอบล่างของการไม่ผ่าน
  it('should return false when score is 1 below threshold (49)', () => {
    expect(isPassScore(49)).toBe(false);
  });

  // 4. Custom threshold
  it('should use custom passThreshold when provided', () => {
    expect(isPassScore(55, 60)).toBe(false); // 55 < 60
    expect(isPassScore(60, 60)).toBe(true);  // 60 >= 60
  });

  // 5. Edge case — score = 0
  it('should return false when score = 0', () => {
    expect(isPassScore(0)).toBe(false);
  });

  // 6. Edge case — score = 100
  it('should return true when score = 100', () => {
    expect(isPassScore(100)).toBe(true);
  });

  // 7. Error — string input
  it('should throw Error when score is a string', () => {
    expect(() => isPassScore('abc')).toThrow(Error);
  });

  // 8. Error — negative score
  it('should throw Error when score is negative', () => {
    expect(() => isPassScore(-1)).toThrow('Invalid score');
  });
});

อ่านฟังก์ชัน getTopScorer() ด้านล่าง แล้ว เขียน Test Suite ทั้งหมด ด้วยตัวเอง ไฟล์ test ควรมีอย่างน้อย 6 test cases ครอบคลุมทั้ง normal และ edge cases

src/utils/scores.js — อ่านแล้วเขียน test
/**
 * หาผู้ที่ได้คะแนนสูงสุดจาก array ของ objects
 * @param {Array} students - Array ของ {name: string, score: number}
 * @returns {object} student object ที่มีคะแนนสูงสุด
 * @throws {Error} ถ้า students ว่างหรือไม่ใช่ Array
 *
 * ตัวอย่าง:
 *   getTopScorer([{name:'A', score:80}, {name:'B', score:90}])
 *   → {name:'B', score:90}
 *
 * กรณีคะแนนเท่ากัน คืน student ตัวแรกที่พบ
 */
function getTopScorer(students) {
  if (!Array.isArray(students) || students.length === 0) {
    throw new Error('students ต้องเป็น Array ที่ไม่ว่าง');
  }
  return students.reduce((top, current) =>
    current.score > top.score ? current : top
  );
}

module.exports = { getTopScorer };

เพิ่ม endpoint ใหม่ GET /api/class-summary ใน app.js แล้ว เขียน test ให้ครอบคลุม ทั้ง happy path และ error cases

1. เพิ่ม Endpoint ใน src/app.js

src/app.js — เพิ่มต่อจากเดิม
// GET /api/class-summary?scores=85,72,60,90,45
// รับคะแนนเป็น comma-separated string
// คืนค่า: { count, average, highest, lowest, gradeDistribution }
app.get('/api/class-summary', (req, res) => {
  const { scores: scoresStr } = req.query;

  if (!scoresStr) {
    return res.status(400).json({ error: 'ต้องส่ง scores parameter' });
  }

  const scores = scoresStr.split(',').map(Number);

  if (scores.some(isNaN)) {
    return res.status(400).json({ error: 'scores ต้องเป็นตัวเลขทั้งหมด' });
  }

  const avg = average(scores);
  const highest = Math.max(...scores);
  const lowest = Math.min(...scores);

  const gradeDistribution = { A: 0, B: 0, C: 0, D: 0, F: 0 };
  scores.forEach(s => {
    const g = getGrade(s);
    gradeDistribution[g]++;
  });

  res.json({
    count: scores.length,
    average: avg,
    highest,
    lowest,
    gradeDistribution,
  });
});

describe('GET /api/class-summary', () => {

  it('คืน summary ที่ถูกต้องจาก scores ปกติ', async () => {
    const res = await request(app)
      .get('/api/class-summary?scores=85,72,60,90,45');
