แยก app.js ออกจาก index.js — เตรียม Testable App
⏱ 8 นาที

ก่อนจะทดสอบ API ด้วย Supertest เราต้อง แยก Express app ออกจาก server — Supertest ต้องการ import app โดยไม่ให้มันเปิด port จริงๆ

⚠️
ปัญหาถ้าไม่แยกถ้า app.listen() อยู่ใน app.js เมื่อ Supertest import ไฟล์นี้ server จะ start จริงและอาจเกิด port conflict หรือ test ค้างไม่จบ

โครงสร้างที่ถูกต้อง

scorecalc/ ├── src/ │ ├── utils/ │ │ ├── calculator.js │ │ └── scorebook.js │ ├── app.js ← Express app (ไม่มี listen) │ └── index.js ← จุดเริ่มต้น (import app แล้ว listen) └── tests/ ├── calculator.test.js ├── scorebook.test.js └── api.test.js ← จะสร้างใน Step 12

1. สร้าง src/app.js — ใส่ Express routes ที่นี่

src/app.js
const express = require('express');
const { getGrade, average } = require('./utils/calculator');

const app = express();
app.use(express.json());

// สถานะ API
app.get('/health', (req, res) => {
  res.json({ status: 'ok', version: '1.0.0' });
});

// GET /api/grade?score=85 — แปลงคะแนนเป็นเกรด
app.get('/api/grade', (req, res) => {
  const score = parseFloat(req.query.score);

  if (isNaN(score)) {
    return res.status(400).json({
      error: 'score ต้องเป็นตัวเลข'
    });
  }

  if (score < 0 || score > 100) {
    return res.status(422).json({
      error: 'score ต้องอยู่ระหว่าง 0-100'
    });
  }

  const grade = getGrade(score);
  res.json({ score, grade });
});

// POST /api/average — คำนวณคะแนนเฉลี่ยจาก array
// Body: { "scores": [85, 90, 72] }
app.post('/api/average', (req, res) => {
  const { scores } = req.body;

  if (!scores || !Array.isArray(scores)) {
    return res.status(400).json({
      error: 'ต้องส่ง scores เป็น Array'
    });
  }

  if (scores.length === 0) {
    return res.status(400).json({
      error: 'scores ต้องมีอย่างน้อย 1 ค่า'
    });
  }

  const avg = average(scores);
  const grade = getGrade(avg);
  res.json({ scores, average: avg, grade });
});

// 404 สำหรับ route ที่ไม่มี
app.use((req, res) => {
  res.status(404).json({ error: 'ไม่พบ endpoint ที่ร้องขอ' });
});

// ส่ง app ออกไป โดยไม่ .listen()
module.exports = app;

2. สร้าง src/index.js — จุดเริ่มต้น Server จริง

src/index.js
const app = require('./app');
const PORT = process.env.PORT || 3000;

app.listen(PORT, () => {
  console.log(`ScoreCalc API รันที่ http://localhost:${PORT}`);
});
✅
ทำไมแยกแบบนี้?
  • app.js export เฉพาะ Express instance — Supertest import ได้โดยไม่ start server
  • index.js เป็นจุดเริ่มต้นจริง — production ใช้ node src/index.js
  • Tests ใช้ require('./src/app') ได้โดยตรง

ติดตั้ง Supertest + เขียน Test GET Endpoint
⏱ 10 นาที

Supertest คือ library ที่ให้เราส่ง HTTP request จริงๆ ไปยัง Express app ใน test โดยไม่ต้องรัน server ขึ้นมาจริงๆ

📌
Supertest ถูกติดตั้งไปแล้วใน Step 3(npm install --save-dev supertest) ไม่ต้องติดตั้งใหม่

สร้างไฟล์ tests/api.test.js

tests/api.test.js
const request = require('supertest');
const app = require('../src/app');

// ============================
// GET /health
// ============================
describe('GET /health', () => {
  it('ต้องตอบ 200 และมี status: ok', async () => {
    // Arrange: ไม่ต้องเตรียมอะไร endpoint นี้ไม่รับ input

    // Act: ส่ง GET request ไปที่ /health
    const res = await request(app).get('/health');

    // Assert: ตรวจสอบ response
    expect(res.status).toBe(200);
    expect(res.body.status).toBe('ok');
    expect(res.body).toHaveProperty('version');
  });
});

// ============================
// GET /api/grade
// ============================
describe('GET /api/grade', () => {

  it('ส่ง score=85 ต้องได้เกรด A', async () => {
    // Act
    const res = await request(app).get('/api/grade?score=85');

    // Assert
    expect(res.status).toBe(200);
    expect(res.body.score).toBe(85);
    expect(res.body.grade).toBe('A');
  });

  it('ส่ง score=72 ต้องได้เกรด B', async () => {
    const res = await request(app).get('/api/grade?score=72');
    expect(res.status).toBe(200);
    expect(res.body.grade).toBe('B');
  });

  it('ส่ง score=0 ต้องได้เกรด F', async () => {
    const res = await request(app).get('/api/grade?score=0');
    expect(res.status).toBe(200);
    expect(res.body.grade).toBe('F');
  });

  it('ส่ง score=100 ต้องได้เกรด A', async () => {
    const res = await request(app).get('/api/grade?score=100');
    expect(res.status).toBe(200);
    expect(res.body.grade).toBe('A');
  });

});

วิธีอ่าน Supertest
  • request(app) — สร้าง test client สำหรับ app ของเรา
  • .get('/path') — ส่ง GET request
  • .post('/path') — ส่ง POST request
  • .send({...}) — ส่ง request body (สำหรับ POST/PUT)
  • res.status — HTTP status code (200, 400, 404...)
  • res.body — response body (parse JSON อัตโนมัติ)

Test POST Endpoint + ตรวจสอบ Response Body
⏱ 10 นาที

สำหรับ POST request เราต้องใช้ .send() เพื่อส่ง request body และตั้งค่า Content-Type ด้วย

เพิ่ม Test สำหรับ POST /api/average ใน tests/api.test.js

tests/api.test.js — เพิ่มต่อจากเดิม
// ============================
// POST /api/average
// ============================
describe('POST /api/average', () => {

  it('คำนวณคะแนนเฉลี่ยและเกรดได้ถูกต้อง', async () => {
    // Arrange
    const scores = [80, 90, 70];
    const expectedAvg = 80;
    const expectedGrade = 'A';

    // Act: ส่ง POST พร้อม body
    const res = await request(app)
      .post('/api/average')
      .send({ scores })           // ส่ง request body
      .set('Content-Type', 'application/json'); // กำหนด Content-Type

    // Assert
    expect(res.status).toBe(200);
    expect(res.body.average).toBe(expectedAvg);
    expect(res.body.grade).toBe(expectedGrade);
  });

  it('response body ต้องมี fields ครบ: scores, average, grade', async () => {
    const res = await request(app)
      .post('/api/average')
      .send({ scores: [75, 85] })
      .set('Content-Type', 'application/json');

    expect(res.status).toBe(200);
    expect(res.body).toHaveProperty('scores');
    expect(res.body).toHaveProperty('average');
    expect(res.body).toHaveProperty('grade');
  });

  it('คำนวณทศนิยมได้ถูกต้อง', async () => {
    const res = await request(app)
      .post('/api/average')
      .send({ scores: [70, 71] });

    expect(res.status).toBe(200);
    expect(res.body.average).toBe(70.5);
  });

  it('คะแนนเฉลี่ย 80 ต้องได้เกรด A', async () => {
    const res = await request(app)
      .post('/api/average')
      .send({ scores: [80, 80, 80] });

    expect(res.body.grade).toBe('A');
  });

});
💡
Shorthand ของ Supertestถ้า app ใช้ express.json() middleware แล้ว Supertest จะ set Content-Type เป็น application/json อัตโนมัติเมื่อใช้ .send(object) — ไม่ต้อง .set('Content-Type', ...) ทุกครั้ง

Test Error Cases — 400, 404, 422
⏱ 9 นาที

การทดสอบ Error Cases มีความสำคัญมากพอๆ กับ happy path — เราต้องมั่นใจว่า API ส่ง error response ที่ถูกต้องกลับไป

เพิ่ม Error Case Tests ใน tests/api.test.js

tests/api.test.js — Error Cases
// ============================
// Error Cases: GET /api/grade
// ============================
describe('GET /api/grade — Error Cases', () => {

  it('ไม่ส่ง score ต้องได้ 400', async () => {
    const res = await request(app).get('/api/grade');
    expect(res.status).toBe(400);
    expect(res.body).toHaveProperty('error');
  });

  it('ส่ง score เป็น string ต้องได้ 400', async () => {
    const res = await request(app).get('/api/grade?score=abc');
    expect(res.status).toBe(400);
    expect(res.body.error).toContain('ตัวเลข');
  });

  it('ส่ง score=-1 ต้องได้ 422 (Unprocessable)', async () => {
    const res = await request(app).get('/api/grade?score=-1');
    expect(res.status).toBe(422);
    expect(res.body).toHaveProperty('error');
  });

  it('ส่ง score=101 ต้องได้ 422', async () => {
    const res = await request(app).get('/api/grade?score=101');
    expect(res.status).toBe(422);
  });

});

// ============================
// Error Cases: POST /api/average
// ============================
describe('POST /api/average — Error Cases', () => {

  it('ไม่ส่ง body ต้องได้ 400', async () => {
    const res = await request(app)
      .post('/api/average')
      .send({});
    expect(res.status).toBe(400);
    expect(res.body).toHaveProperty('error');
  });

  it('ส่ง scores เป็น string แทน Array ต้องได้ 400', async () => {
    const res = await request(app)
      .post('/api/average')
      .send({ scores: 'ไม่ใช่ array' });
    expect(res.status).toBe(400);
  });

  it('ส่ง scores เป็น array ว่างต้องได้ 400', async () => {
    const res = await request(app)
      .post('/api/average')
      .send({ scores: [] });
    expect(res.status).toBe(400);
    expect(res.body.error).toContain('อย่างน้อย 1');
  });

});

// ============================
// 404 Not Found
// ============================
describe('404 — Route ที่ไม่มี', () => {

  it('GET /nonexistent ต้องได้ 404', async () => {
    const res = await request(app).get('/nonexistent');
    expect(res.status).toBe(404);
    expect(res.body).toHaveProperty('error');
  });

  it('POST /unknown ต้องได้ 404', async () => {
    const res = await request(app).post('/unknown');
    expect(res.status).toBe(404);
  });

});
🎯
HTTP Status Codes ที่ควรรู้
  • 200OK — สำเร็จ
  • 201Created — สร้าง resource ใหม่สำเร็จ
  • 400Bad Request — input ไม่ถูกต้อง
  • 404Not Found — ไม่พบ endpoint หรือ resource
  • 422Unprocessable — format ถูกแต่ค่าไม่ valid
  • 500Internal Server Error — bug ใน server

Code Coverage Report + ตั้งค่า Threshold
⏱ 8 นาที

Code Coverage วัดว่าโค้ดของเราถูก "ผ่าน" (executed) โดย test กี่เปอร์เซ็นต์ — ช่วยหาส่วนของโค้ดที่ยังไม่มี test คุ้มครอง

1. รัน Coverage Report

Terminal
npm run test:coverage

2. ผลลัพธ์ที่จะได้

------------------------|---------|----------|---------|---------|
All files | 88.09 | 83.33 | 87.50 | 88.09 |

3. ความหมายของแต่ละ Column

  • % Stmts (Statements): กี่ % ของ statement ทั้งหมดถูกรัน
  • % Branch: กี่ % ของ if/else branch ถูกทดสอบ
  • % Funcs: กี่ % ของ function ถูกเรียกใช้
  • % Lines: กี่ % ของ line โค้ดถูกรัน

4. ตั้งค่า Coverage Threshold (ขั้นต่ำ)

เปิด package.json ตรวจสอบให้มี coverageThresholds — Jest จะ fail หาก coverage ต่ำกว่าที่กำหนด:

package.json — coverageThresholds (ควรมีอยู่แล้วจาก Step 3)
{
  "jest": {
    "coverageThresholds": {
      "global": {
        "branches": 70,
        "functions": 70,
        "lines": 70,
        "statements": 70
      }
    }
  }
}
