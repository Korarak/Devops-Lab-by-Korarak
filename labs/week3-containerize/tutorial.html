week3-containerize/
├── backend/                # Express.js API Server
│   ├── index.js            # routes, initDb(), seed data
│   ├── db.js               # PostgreSQL connection pool
│   ├── package.json        # express, cors, pg, dotenv
│   ├── .env.example        # template สำหรับ env vars
│   ├── Dockerfile          # build backend image
│   └── .dockerignore
│
├── frontend/               # Vue 3 + Vite SPA
│   ├── src/
│   │   ├── App.vue         # root component: รวม components ทั้งหมด
│   │   ├── main.js         # entry point: mount App เข้า #app
│   │   ├── composables/
│   │   │   └── useProducts.js  # state + computed + API methods ทั้งหมด
│   │   └── components/
│   │       ├── AppHeader.vue       # header + ปุ่มเพิ่มสินค้า
│   │       ├── StatsGrid.vue       # dashboard 4 card
│   │       ├── LowStockAlert.vue   # แจ้งเตือนสต็อกใกล้หมด
│   │       ├── ProductToolbar.vue  # search + filter
│   │       ├── ProductGrid.vue     # grid wrapper + loading/empty states
│   │       ├── ProductCard.vue     # card สินค้าแต่ละชิ้น
│   │       ├── ProductModal.vue    # modal เพิ่ม/แก้ไขสินค้า
│   │       └── DeleteConfirm.vue   # dialog ยืนยันการลบ
│   ├── index.html          # HTML template (Vite inject script ที่นี่)
│   ├── vite.config.js      # proxy /api → backend:3001
│   ├── package.json        # vue, vite, @vitejs/plugin-vue
│   ├── nginx.conf          # Nginx config (production proxy)
│   ├── Dockerfile          # multi-stage: build Vue → serve Nginx
│   └── .dockerignore
│
├── docker-compose.yml      # dev: build จาก source code
├── docker-compose.prod.yml # prod: pull จาก Docker Hub
└── .gitignore

✅ สิ่งที่ต้องมีก่อนเริ่ม

  • Node.js 18+ — ตรวจสอบด้วย node -v
  • Docker Desktop รันอยู่ — ตรวจสอบด้วย docker info
  • Git — สำหรับ clone repo
  • บัญชี Docker Hub (ฟรี) ที่ hub.docker.com — ใช้ใน Part 3
  • อยู่ใน folder week3-containerize/ ก่อนเริ่มรัน command ใดๆ
