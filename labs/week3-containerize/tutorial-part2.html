composables/useProducts.js — State และ Logic ทั้งหมด
Composable

Composable คือ function ที่ return reactive state และ methods ออกมา — เก็บ logic ไว้ที่เดียว แล้ว component อื่นเรียกใช้ได้

🧩
ทำไมต้องมี Composable? ถ้าใส่ state + API calls ไว้ใน App.vue → App.vue ยาว 200+ บรรทัด
แยกออกมาเป็น useProducts.js → App.vue เหลือแค่ import component + เรียก composable
เปิดไฟล์: frontend/src/composables/useProducts.js~80 บรรทัด

ส่วนที่ 1 — State (ref):

useProducts.js — State
import { ref, computed, watch } from 'vue'

export function useProducts() {

  // ── State ────────────────────────────────────────────────────────
  const products      = ref([])        // รายการสินค้าจาก API
  const loading       = ref(true)      // กำลังโหลดอยู่หรือไม่
  const search        = ref('')        // ข้อความค้นหา
  const catFilter     = ref('')        // category ที่เลือก
  const showModal     = ref(false)     // แสดง/ซ่อน modal เพิ่ม/แก้ไข
  const editingId     = ref(null)      // null = Add, มีค่า = Edit
  const confirmDelete = ref(null)      // product ที่กำลังจะลบ

  const emptyForm = () => ({ name: '', category: '', price: '', stock: 0, description: '' })
  const form = ref(emptyForm())        // ข้อมูลที่ส่งให้ ProductModal
ref([])สร้าง reactive array — เมื่อ products.value เปลี่ยน Vue re-render ส่วนที่ใช้ products อัตโนมัติ
editingIdnull = mode Add (POST), มีค่า = mode Edit (PUT) — ใช้ตัดสินใจใน saveProduct
emptyForm()function สร้าง object ว่างสำหรับ form ใหม่ — เรียกทุกครั้งที่เปิด modal Add

ส่วนที่ 2 — Computed:

useProducts.js — Computed
  // ── Computed ─────────────────────────────────────────────────────
  const filtered = computed(() => {
    const s = search.value.toLowerCase()
    return products.value.filter(p => {
      const matchS = !s || p.name.toLowerCase().includes(s) ||
                     (p.description || '').toLowerCase().includes(s)
      const matchC = !catFilter.value || p.category === catFilter.value
      return matchS && matchC
    })
  })

  const categories = computed(() =>
    [...new Set(products.value.map(p => p.category))].sort()
  )

  const stats = computed(() => ({
    total:      products.value.length,
    lowStock:   products.value.filter(p => p.stock < 10).length,
    totalItems: products.value.reduce((s, p) => s + p.stock, 0),
    totalValue: products.value.reduce((s, p) => s + parseFloat(p.price) * p.stock, 0)
  }))
computed()คำนวณค่าจาก state — Vue cache ผลไว้, คำนวณใหม่เฉพาะเมื่อ dependencies (products, search, catFilter) เปลี่ยน
filteredกรองสินค้าตาม search text AND category — ProductGrid ใช้ค่านี้แสดงผล
new Set(...)ดึง category unique จาก products ทั้งหมด — ตัดซ้ำ, sort ตัวอักษร

ส่วนที่ 3 — Methods:

useProducts.js — API Methods
  // ── API Methods ───────────────────────────────────────────────────
  async function fetchProducts() {
    loading.value = true
    const q = new URLSearchParams()
    if (search.value)    q.set('search',   search.value)
    if (catFilter.value) q.set('category', catFilter.value)
    try {
      const res = await fetch(`/api/products?${q}`)
      products.value = await res.json()
    } catch {
      products.value = []
    } finally {
      loading.value = false
    }
  }

  function openAdd() {
    editingId.value = null
    form.value = emptyForm()
    showModal.value = true
  }

  function openEdit(p) {
    editingId.value = p.id
    form.value = { name: p.name, category: p.category,
                   price: p.price, stock: p.stock, description: p.description || '' }
    showModal.value = true
  }

  async function saveProduct(formData) {
    const body = { name: formData.name, category: formData.category,
                   price: parseFloat(formData.price), stock: parseInt(formData.stock),
                   description: formData.description }
    const url    = editingId.value ? `/api/products/${editingId.value}` : '/api/products'
    const method = editingId.value ? 'PUT' : 'POST'
    await fetch(url, { method, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) })
    showModal.value = false
    fetchProducts()
  }

  async function deleteProduct(id) {
    await fetch(`/api/products/${id}`, { method: 'DELETE' })
    confirmDelete.value = null
    fetchProducts()
  }

ส่วนที่ 4 — Watchers + Return:

useProducts.js — Watch + Return
  // ── Watchers ──────────────────────────────────────────────────────
  let debounceTimer = null
  watch(search, () => {
    clearTimeout(debounceTimer)
    debounceTimer = setTimeout(fetchProducts, 280)  // รอ 280ms หลังพิมพ์หยุด
  })

  watch(catFilter, fetchProducts)  // fetch ทันทีเมื่อเลือก category

  return {
    products, loading, search, catFilter, showModal, editingId, confirmDelete, form,
    filtered, categories, stats,
    fetchProducts, openAdd, openEdit, saveProduct, deleteProduct
  }
}
watch(search, ...)ทุกครั้งที่ search เปลี่ยน → clear timer เก่า → set timer ใหม่ 280ms → fetchProducts
ถ้าพิมพ์ต่อก่อน 280ms → timer รีเซ็ต → ไม่ fetch ทุก keystroke
watch(catFilter, ...)catFilter เปลี่ยน → fetchProducts ทันที (ไม่ต้อง debounce เพราะเลือกครั้งเดียว)
return {...}export ทุกอย่างออกมา — App.vue destructure เอาแค่ที่ต้องการ
🔁
Watch vs @input — ต่างกันยังไง? watch(search, ...) ใน composable ตรวจ state เปลี่ยนเอง — component ไม่ต้องส่ง event
ผล: ProductToolbar แค่ update search ผ่าน v-model, composable จัดการ fetch เอง

✅
เข้าใจ Pattern นี้แล้วใช่ไหม?
  • Props = ทางเดียว จาก parent → child ผ่าน :propName="value"
  • Emits = ทางเดียว จาก child → parent ผ่าน $emit('eventName', data)
  • v-model = สะดวกขึ้น: รวม prop + emit เข้าด้วยกัน

ProductCard.vue — <script setup>
defineProps({ product: { type: Object, required: true } })

// คืน CSS class ตามระดับ stock
function stockClass(s) {
  if (s <= 0) return 'out'   // หมด
  if (s < 10) return 'low'   // ใกล้หมด
  if (s < 30) return 'mid'   // ปานกลาง
  return 'high'              // เพียงพอ
}

// คืน CSS class ตาม category (ใช้กับ badge สี)
function catClass(cat) {
  const m = { Electronics: 'c-elec', Clothing: 'c-cloth', ... }
  return m[cat] || 'c-other'
}
ProductCard.vue — template (ส่วนหลัก)
<!-- :class dynamic — ขอบแดงถ้า stock ใกล้หมด -->
<div class="product-card"
  :class="{ 'card-low': product.stock > 0 && product.stock < 10,
             'card-out': product.stock <= 0 }">

  <span class="cat-badge" :class="catClass(product.category)">
    {{ product.category }}
  </span>
  <div class="product-name">{{ product.name }}</div>
  <div class="product-price">
    ฿{{ parseFloat(product.price).toLocaleString('th-TH', {minimumFractionDigits: 2}) }}
  </div>

  <!-- progress bar ตามระดับ stock -->

    :class="stockClass(product.stock)"
    :style="{ width: Math.min((product.stock / 80) * 100, 100) + '%' }">

3. ProductModal.vue — Form เพิ่ม/แก้ไขสินค้า

เปิดไฟล์: frontend/src/components/ProductModal.vue~80 บรรทัด
ProductModal.vue — <script setup>
import { reactive } from 'vue'

const props = defineProps({
  editingId:   { default: null },
  initialForm: { type: Object, required: true },
  categories:  { type: Array,  required: true },
})
const emit = defineEmits(['save', 'close'])

// copy ค่าเริ่มต้นจาก props เข้า local state — ไม่แก้ props โดยตรง
const localForm = reactive({ ...props.initialForm })

function onSubmit() {
  emit('save', { ...localForm })  // ส่ง form data กลับ App.vue
}
ProductModal.vue — template (form)
<!-- @click.self — ปิด modal เมื่อคลิก overlay แต่ไม่ปิดเมื่อคลิกใน modal -->
<div class="overlay" @click.self="$emit('close')">
  <div class="modal">
    <div class="modal-title">
      {{ editingId ? '✏️ แก้ไขสินค้า' : '📦 เพิ่มสินค้าใหม่' }}
    </div>
    <form @submit.prevent="onSubmit">
      <input v-model="localForm.name" ... required />
      <input v-model="localForm.category" ... required />
      <input v-model="localForm.price"    type="number" ... required />
      <input v-model="localForm.stock"    type="number" ... required />
      <textarea v-model="localForm.description" ...></textarea>
      <button type="submit">{{ editingId ? 'บันทึก' : 'เพิ่มสินค้า' }}</button>
    </form>
  </div>
</div>
reactive({ ...props.initialForm })spread copy props → ได้ object ใหม่ที่ไม่เชื่อมกับ props — แก้ได้อิสระใน form
@submit.prevent.prevent = preventDefault() — หยุด form จากการ reload หน้า
emit('save', { ...localForm })ส่ง form data กลับ App.vue → saveProduct(formData) ใน composable รับต่อ

import ProductGrid       from './components/ProductGrid.vue'
import ProductModal      from './components/ProductModal.vue'
import DeleteConfirm     from './components/DeleteConfirm.vue'

// destructure ทุกอย่างจาก composable
const {
  products, loading, search, catFilter, showModal, editingId, confirmDelete, form,
  filtered, categories, stats,
  fetchProducts, openAdd, openEdit, saveProduct, deleteProduct
} = useProducts()

onMounted(fetchProducts)   // โหลดสินค้าครั้งแรกเมื่อ App mount
App.vue — <template>
<template>
  <div>
    <AppHeader @add="openAdd" />

    <main class="main">
      <StatsGrid     :stats="stats" />
      <LowStockAlert :count="stats.lowStock" />
      <ProductToolbar
        v-model:search="search"
        v-model:catFilter="catFilter"
        :categories="categories"
        :loading="loading"
        :filtered-count="filtered.length"
        :total-count="products.length"
      />
      <ProductGrid
        :products="filtered"
        :loading="loading"
        :is-filtered="!!(search || catFilter)"
        @edit="openEdit"
        @request-delete="p => (confirmDelete = p)"
      />
    </main>

    <!-- v-if: สร้าง modal เฉพาะเมื่อจำเป็น → mount ใหม่ทุกครั้ง -->
    <ProductModal
      v-if="showModal"
      :editing-id="editingId"
      :initial-form="form"
      :categories="categories"
      @save="saveProduct"
      @close="showModal = false"
    />
    <DeleteConfirm
      v-if="confirmDelete"
      :product="confirmDelete"
      @confirm="deleteProduct(confirmDelete.id)"
      @cancel="confirmDelete = null"
    />
  </div>
</template>

อธิบาย data flow ทั้งระบบ:

@add="openAdd"AppHeader กดปุ่ม → emit 'add' → App.vue เรียก openAdd() → composable set showModal=true
v-model:search="search"ProductToolbar พิมพ์ → emit update:search → search ref เปลี่ยน → watch → debounce → fetchProducts
@request-delete="p => confirmDelete = p"กดลบใน ProductCard → ProductGrid re-emit → App.vue set confirmDelete = product → DeleteConfirm แสดง
@save="saveProduct"กด submit ใน ProductModal → emit 'save' พร้อม formData → App.vue เรียก saveProduct(formData)
v-if="showModal"mount ProductModal ใหม่ทุกครั้งที่เปิด → localForm ใน Modal reset เสมอ
