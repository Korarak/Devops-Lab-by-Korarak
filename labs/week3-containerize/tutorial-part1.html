backend/db.js — PostgreSQL Connection Pool
Database

เปิดไฟล์ backend/db.js — ไฟล์นี้สร้าง connection pool ไปยัง PostgreSQL และ export ออกให้ index.js ใช้

เปิดไฟล์: backend/db.js
backend/db.js — เนื้อหาทั้งหมด
const { Pool } = require('pg');
require('dotenv').config();

const pool = new Pool({
  host:     process.env.DB_HOST,
  port:     process.env.DB_PORT,
  database: process.env.DB_NAME,
  user:     process.env.DB_USER,
  password: process.env.DB_PASSWORD,
});

pool.query('SELECT NOW()', (err, res) => {
  if (err) console.error('❌ DB connection failed:', err.message);
  else     console.log('✅ DB connected at:', res.rows[0].now);
});

module.exports = pool;

อธิบายแต่ละส่วน:

require('pg')โหลด library pg (node-postgres) — ใช้สำหรับเชื่อมต่อ PostgreSQL จาก Node.js
require('dotenv').config()โหลดค่าจาก .env เข้า process.env — ทำให้ DB_HOST, DB_PORT ฯลฯ พร้อมใช้
new Pool({...})สร้าง connection pool — Pool จัดการ connections หลายอันพร้อมกัน (ต่างจาก Client เดี่ยว)
process.env.DB_HOSTค่ามาจาก .env ตอน local, มาจาก environment ใน docker-compose ตอน containerize
pool.query('SELECT NOW()')ทดสอบการเชื่อมต่อตอน startup — ถ้า error จะ log ออกมาให้รู้ทันที
module.exports = poolexport pool ออกให้ index.js ใช้ได้ด้วย require('./db')
🔌
Pool vs Client Pool รองรับ concurrent requests — แต่ละ request ดึง connection จาก pool มาใช้แล้วคืน ต่างจาก Client ที่มี connection เดียว ซึ่งจะติด bottleneck ถ้ามีหลาย request พร้อมกัน
🏗️
ทำไม DB_HOST เป็น "db" ใน Docker แต่ "localhost" ตอน local? ใน Docker Compose แต่ละ service คุยกันผ่านชื่อ service — Docker DNS resolve ชื่อ "db" เป็น IP ของ container อัตโนมัติ ตอน local ต้องใช้ "localhost" เพราะ postgres รันบนเครื่องเดียวกัน

backend/index.js — Express API ทั้งหมด
Express

เปิดไฟล์ backend/index.js — ไฟล์นี้มี Express routes ทั้งหมด, สร้างตาราง products อัตโนมัติ, และใส่ seed data

เปิดไฟล์: backend/index.js

ส่วนที่ 1 — Setup:

backend/index.js — บรรทัด 1-11
const express = require('express');
const cors    = require('cors');
const pool    = require('./db');
require('dotenv').config();

const app  = express();
const PORT = process.env.PORT || 3001;

app.use(cors());
app.use(express.json());
require('./db')import pool จาก db.js — connection pool พร้อมใช้งาน
app.use(cors())อนุญาต cross-origin requests — ทำให้ frontend ที่ต่าง port เรียก API ได้
app.use(express.json())parse JSON body — ทำให้ req.body มีข้อมูลจาก POST/PUT request

ส่วนที่ 3 — GET /api/products (พร้อม search + filter):

backend/index.js — GET products
app.get('/api/products', async (req, res) => {
  try {
    const { search = '', category = '' } = req.query;
    const params = [];
    let where = 'WHERE 1=1';

    if (search) {
      params.push(`%${search}%`);
      where += ` AND (name ILIKE $${params.length} OR description ILIKE $${params.length})`;
    }
    if (category) {
      params.push(category);
      where += ` AND category = $${params.length}`;
    }

    const { rows } = await pool.query(
      `SELECT * FROM products ${where} ORDER BY created_at DESC`,
      params
    );
    res.json(rows);
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
});
WHERE 1=1เริ่มด้วย condition ที่เป็น true เสมอ — ทำให้ append AND ... ต่อท้ายได้ง่ายโดยไม่ต้องเช็คว่าเป็น WHERE แรกหรือเปล่า
ILIKE $1ILIKE = case-insensitive LIKE, $1 = parameterized query ป้องกัน SQL Injection
params.push()เพิ่มค่าลง array params แล้วใช้ $1, $2 แทน — pg library inject ให้อัตโนมัติ

ส่วนที่ 4 — POST / PUT / DELETE:

backend/index.js — CRUD endpoints
// POST /api/products — เพิ่มสินค้าใหม่
app.post('/api/products', async (req, res) => {
  try {
    const { name, category, price, stock, description = '' } = req.body;
    if (!name || !category || price == null || stock == null) {
      return res.status(400).json({ error: 'กรุณาระบุ name, category, price, stock' });
    }
    const { rows } = await pool.query(
      `INSERT INTO products (name, category, price, stock, description)
       VALUES ($1, $2, $3, $4, $5) RETURNING *`,
      [name.trim(), category.trim(), parseFloat(price), parseInt(stock), description.trim()]
    );
    res.status(201).json(rows[0]);
  } catch (err) { res.status(500).json({ error: err.message }); }
});

// PUT /api/products/:id — แก้ไขสินค้า
app.put('/api/products/:id', async (req, res) => {
  try {
    const { name, category, price, stock, description = '' } = req.body;
    const { rows } = await pool.query(
      `UPDATE products
       SET name=$1, category=$2, price=$3, stock=$4, description=$5, updated_at=NOW()
       WHERE id=$6 RETURNING *`,
      [name, category, parseFloat(price), parseInt(stock), description, req.params.id]
    );
    if (!rows[0]) return res.status(404).json({ error: 'ไม่พบสินค้า' });
    res.json(rows[0]);
  } catch (err) { res.status(500).json({ error: err.message }); }
});

// DELETE /api/products/:id — ลบสินค้า
app.delete('/api/products/:id', async (req, res) => {
  try {
    const { rows } = await pool.query(
      'DELETE FROM products WHERE id=$1 RETURNING *', [req.params.id]
    );
    if (!rows[0]) return res.status(404).json({ error: 'ไม่พบสินค้า' });
    res.json({ message: 'ลบสินค้าสำเร็จ', deleted: rows[0] });
  } catch (err) { res.status(500).json({ error: err.message }); }
});

ส่วนที่ 5 — initDb() สร้างตารางและ seed data:

backend/index.js — initDb() + app.listen()
async function initDb() {
  // สร้างตาราง products ถ้ายังไม่มี
  await pool.query(`
    CREATE TABLE IF NOT EXISTS products (
      id          SERIAL PRIMARY KEY,
      name        VARCHAR(150) NOT NULL,
      category    VARCHAR(80)  NOT NULL,
      price       NUMERIC(12,2) NOT NULL DEFAULT 0,
      stock       INTEGER      NOT NULL DEFAULT 0,
      description TEXT         DEFAULT '',
      created_at  TIMESTAMPTZ  DEFAULT NOW(),
      updated_at  TIMESTAMPTZ  DEFAULT NOW()
    )
  `);

  // Seed data ถ้า table ยังว่าง

// รัน initDb ก่อน start server
initDb()
  .then(() => {
    app.listen(PORT, () => {
      console.log(`🚀 StockPro API running at http://localhost:${PORT}`);
    });
  })
  .catch(err => {
    console.error('❌ DB init failed:', err.message);
